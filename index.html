Simple version


// Notice
Growl.Notice("Your message has been sent.");
// Warning
Growl.Warn("Your account will expire in 2 days.");
// Error
Growl.Error("You should login first.");

Not so complex version


// Notice
Growl.Notice(
  {
    message: "Your message has been sent.", // The message to be shown
    timeout: 5000, // Time before the message disappears, in ms,
    title: "Congratulations" // Title of the message
  }
);
// Warning
Growl.Warn(
  {
    message: "Your account will expire in 2 days!", // The message to be shown
    timeout: 5000, // Time before the message disappears, in ms,
    title: "Attention" // Title of the message
  }
);
// Error
Growl.Error(
  {
    message: "You don't have permission to access this area!", // The message to be shown
    timeout: 5000, // Time before the message disappears, in ms,
    title: "Error" // Title of the message
  }
);
